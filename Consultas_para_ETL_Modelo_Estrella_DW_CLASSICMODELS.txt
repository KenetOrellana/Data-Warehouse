---------- CONSULTA PARA LA TABLA TIEMPO (TBL_TIEMPO) ------------
SELECT CONVERT(DATE, ORDERDATE) TIEMPO_ID, DATENAME(MONTH, ORDERDATE) NOMBRE_MES, DATEPART(WEEK, ORDERDATE) SEMANA, DATEPART(QUARTER, ORDERDATE) TRIMESTRE, CEILING(DATEPART(MM, ORDERDATE)*1.0/4) CUATRIMESTRE, CEILING(DATEPART(MM, ORDERDATE)/6) SEMESTRE, DATENAME(WEEKDAY, ORDERDATE) DIA_DE_SEMANA FROM ORDERS

GROUP BY ORDERS.ORDERDATE

---------- CONSULTA PARA LA TABLA PRODUCTOS (PRODUCTS) ------------
SELECT PRODUCTCODE, PRODUCTNAME, PRODUCTLINE, QUANTITYINSTOCK FROM PRODUCTS

---------- CONSULTA PARA LA TABLA EMPLEADOS (EMPLOYEES)  ------------
SELECT EMPLOYEENUMBER, CONCAT(FIRSTNAME,' ', LASTNAME) NOMBRE_EMPLOYEE, OFFICECODE FROM EMPLOYEES

---------- CONSULTA PARA LA TABLA CLIENTES (CUSTOMERS) ------------
SELECT CUSTOMERNUMBER, CUSTOMERNAME, CONCAT(CONTACTFIRSTNAME,' ',CONTACTLASTNAME) NOMBRE_CONTACTO_CUSTOMER, CITY, COUNTRY FROM CUSTOMERS

---------- CONSULTA PARA LA TABLA HECHOS_CLASSICMODELS ------------
SELECT EMPLOYEES.EMPLOYEENUMBER, CUSTOMERS.CUSTOMERNUMBER, PRODUCTS.PRODUCTCODE, CONVERT(DATE, ORDERDATE) TIEMPO_ID,

SUM([ORDERDETAILS].QUANTITYORDERED*[ORDERDETAILS].PRICEEACH) TOTAL_VENTA_PRODUCTO FROM ORDERS
INNER JOIN ORDERDETAILS ON ORDERS.ORDERNUMBER = ORDERDETAILS.ORDERNUMBER
INNER JOIN EMPLOYEES ON EMPLOYEES.EMPLOYEENUMBER = ORDERS.EMPLOYEENUMBER
INNER JOIN CUSTOMERS ON CUSTOMERS.CUSTOMERNUMBER = ORDERS.CUSTOMERNUMBER
INNER JOIN PRODUCTS ON PRODUCTS.PRODUCTCODE = ORDERDETAILS.PRODUCTCODE

GROUP BY EMPLOYEES.EMPLOYEENUMBER, CUSTOMERS.CUSTOMERNUMBER, PRODUCTS.PRODUCTCODE, ORDERDATE
